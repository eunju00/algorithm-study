```javascript
const readline = require('readline');

const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout,
});

rl.on('line', function(line) {

    let words = line.trim().split(' ');


    if (words[0] === '') {
        console.log(0);
    } else {
        console.log(words.length);
    }

    rl.close();
}).on('close', function() {
    process.exit();
});
```
